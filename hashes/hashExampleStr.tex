    \item insert \myemph<2>{``foo''}, \myemph<3>{``bar''},\myemph<4>{``baz''}

            \item<4-> \myemph<4>{h(``baz'') = 317 --- bucket $317 \mod 11 = 9$ }
        \end{itemize}
    \item find \myemph<5>{``baz''}, \myemph<6>{``quux''}
            \item<6-> \myemph<6>{h(``quux'') = 317 --- bucket $467 \mod 11 = 5$ }
    \end{itemize}
