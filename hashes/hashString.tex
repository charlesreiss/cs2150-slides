\begin{frame}[fragile,label=hashStrProposals1]{some proposals (1)}

unsigned int hash(const string &s) {

}

unsigned int hash(const string &s) {
    unsigned int sum = 0;
    for (int i = 0; i < s.size(); ++i) {
        sum += s[i];
    }
    return sum;
}
