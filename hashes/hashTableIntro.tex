\item array of some size
    \begin{itemize}
    \item larger than \# of total elements
    \item usually prime size
    \end{itemize}

\end{itemize}

    \draw[decorate,decoration={brace,mirror},draw,thick] (hashtable.north west) -- (hashtable.south west);

\begin{itemize}
\item \textit{must be} \myemph{deterministic}
    \begin{itemize}
    \item<1-> each key assigned to exactly one ``bucket''
    \end{itemize}
\item<2-> \textit{should be} \myemph{evenly distributed}
    \begin{itemize}
    \item<2-> two keys \textit{unlikely} to share bucket
    \item<2-> each bucket about as used as each other bucket
    \end{itemize}
\item<3-> \textit{should be} fast
\end{itemize}
