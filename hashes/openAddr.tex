\begin{frame}[fragile,label=avoidLL]{avoiding linked lists}
\begin{lstlisting}
class HashTableBucket {
    ...
    int key;               // 4 bytes
    int value;             // 4 bytes
    HashTableBucket *next; // 8 bytes
};
\end{lstlisting}
\begin{itemize}
\item gosh, that's a lot of overhead
\item \ldots even though ``usually'' one item/bucket
\end{itemize}
\end{frame}
