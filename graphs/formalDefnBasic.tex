\begin{frame}{formal definition}
\begin{itemize}
\item graph $G$: $G = (V,E)$
\item $V$: set of vertices (possibly empty)
\item $E$: set of edges --- pairs of vertices (possibly empty)
    \begin{itemize}
    \item directed graph/digraph --- ordered pairs
    \item undirected graph --- unordered pairs
    \end{itemize}
\end{itemize}
\end{frame}
