int max(int x, int y) {
    int theMax;
    if (x > y) 
        theMax = x;
    else
        theMax = y;
    return theMax;
}

    int maxVal, a = 5, b = 6;
    maxVal = max(a, b);
    cout << "max value: " << maxVal << endl;
    return 0;
