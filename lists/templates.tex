\begin{frame}{templates}
    \begin{itemize}
    \item \textit{templates} --- C++'s equivalent to \textit{generics}
    \vspace{.5cm}
    \item idea --- code with `fill in the blank'
    \item compiler genreates seperate version for each blank
    \end{itemize}
\end{frame}
