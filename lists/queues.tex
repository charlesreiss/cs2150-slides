\begin{frame}{queue v stack}
\begin{itemize}
\item queue --- first-in, first-out (FIFO)
\item stack --- last-in, first-out (LIFO)
\vspace{.5cm}
\item both have linked list, array-based implementations
\end{itemize}
\end{frame}
