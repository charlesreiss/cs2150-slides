void binary_heap::percolateUp(int index) {
    int newValue = heap[index];
    // while not at root and
    //       less than parent...
    while (index > 1 && newValue < heap[index / 2]) {
        // move parent down
        heap[index] = heap[index / 2];
        // advanced up the tree
        index /= 2;
    }
    heap[index] = newValue;
}
\end{lstlisting}
