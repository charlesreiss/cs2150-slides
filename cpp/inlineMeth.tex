\begin{frame}[fragile,label=inlineMethod2]{inline methods (2)}
\lstset{
    language=C++,
    style=smaller,
    moredelim={**[is][\btHL<all:2-3>]{@2}{2@}},
    moredelim={**[is][\btHL<all:3>]{@3}{3@}},
}
\begin{lstlisting}
class Foo {
public:
    Foo();
    int getValue() const;
    ...
private:
    int value;
    ...
};
@2inline2@ int Foo::getValue() const {
    return value;
}
\end{lstlisting}

    \textbf{must} be included by every {\tt .cpp} that uses class
};
\end{visibleenv}
\end{tikzpicture}
\end{frame}
